create database userDetails;

use userDetails;

CREATE TABLE users (
     id MEDIUMINT NOT NULL AUTO_INCREMENT,
     username VARCHAR(45) NOT NULL,
    password VARCHAR(45) NOT NULL,
    enabled INT NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE authorities (
     id MEDIUMINT NOT NULL AUTO_INCREMENT,
     username VARCHAR(45) NOT NULL,
    authority VARCHAR(45) NOT NULL,
     PRIMARY KEY (id)
);
